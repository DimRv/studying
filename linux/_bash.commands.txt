Текстовый файл должен иметь возможность выполнения (x)

Указание на интерпретатор:
#!/bin/bash
#!/bin/sh
#!/usr/bin/perl
#!/usr/bin/python3

Выходные коды в $?:
0 - успешное завершение
1-255 - с ошибкой

# - комментарий
= - создание переменной
$ - чтение переменной
| - отправка результата выполнения команды другой команде
\ - экранирование
1> - стандартный поток
2> - ошибки
> - отправка вывода команды в файл с перезаписью
>> - добавление файла результатом выполнения команды
{01..12} - диапазон от 01 до 12
var = value - создание переменной ($var - чтение переменной)
$1 - $9 - автоматические переменные из аргументов скрипта
unset var - удаление переменной
export - создание временной переменной
alias - создание псевдонима
unalias - отключение псевдонима
echo - простой вывод информации
printf - форматированный вывод
read - ввод данных
history - история вводимых команд
bash -v -x - для отладки скриптов
let - математические действия (+,-,*,/,**,%)
test - используется для сравнений: test "$A" = "$B" ["$A" = "$B"], а также файлы

(условие) - истинно ли условие
! условие - ложно ли условие
усл1 -а усл2 - &&
усл1 -o усл2 - ||
-n str - длина строки не равна 0
-z str - длина строки равна 0
n1 -eq n2 (-ne, -ge, -gt, -le, -lt)
f1 -ef f2 (-nt -новее, -ot - старее

if усл; then
Команда1
else
Команда2
fi

case строка in
Шаблон)
команда1;;
Шаблон)
команда2;;
esac

while условие; do
команда
done

until - выполняется пока условие не станет true

continue - возвращение к do
break - выход из цикла
exit - выход из программы

for усл; do
команды
done
